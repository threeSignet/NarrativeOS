## 1. 断点体系

### 1.1 核心断点定义表

| 断点名 | 像素范围 | 目标设备 | 方向 | 布局策略 | Tailwind前缀 |
|---|---|---|---|---|---|
| `xs` | < 480px | 手机竖屏 | Portrait | 单栏堆叠，底部Tab | 默认（无前缀） |

### 1.2 方向感知断点（Orientation Breakpoints）

```typescript
// src/styles/breakpoints.ts
export const breakpoints = {
  // 基础宽度断点
  xs: 0,      // 手机竖屏
  sm: 480,    // 手机横屏
  md: 768,    // Pad竖屏
  lg: 1024,   // Pad横屏
  xl: 1280,   // PC桌面
  '2xl': 1536,// PC宽屏
} as const;

// 方向感知媒体查询组合
export const orientationQueries = {
  portrait: {
    xs: '(max-width: 479px) and (orientation: portrait)',
    sm: '(max-width: 767px) and (orientation: portrait)',
    md: '(min-width: 768px) and (max-width: 1023px) and (orientation: portrait)',
    lg: '(min-width: 1024px) and (max-width: 1279px) and (orientation: portrait)',
  },
  landscape: {
    sm: '(min-width: 480px) and (max-width: 767px) and (orientation: landscape)',
    md: '(min-width: 768px) and (max-width: 1023px) and (orientation: landscape)',
    lg: '(min-width: 1024px) and (max-width: 1279px) and (orientation: landscape)',
    xl: '(min-width: 1280px) and (orientation: landscape)',
  }
} as const;

// 设备类型判断
export type DeviceType = 'mobile' | 'tablet' | 'desktop';
export type Orientation = 'portrait' | 'landscape';

export const getDeviceType = (width: number): DeviceType => {
  if (width < 768) return 'mobile';
  if (width < 1280) return 'tablet';
  return 'desktop';
};

export const useDeviceProfile = () => {
  const [profile, setProfile] = useState({
    width: window.innerWidth,
    height: window.innerHeight,
    deviceType: getDeviceType(window.innerWidth),
    orientation: window.innerWidth > window.innerHeight ? 'landscape' : 'portrait',
    isTouch: 'ontouchstart' in window,
    isPointer: window.matchMedia('(pointer: fine)').matches,
  });

  useEffect(() => {
    const handler = () => {
      const w = window.innerWidth;
      const h = window.innerHeight;
      setProfile({
        width: w,
        height: h,
        deviceType: getDeviceType(w),
        orientation: w > h ? 'landscape' : 'portrait',
        isTouch: 'ontouchstart' in window,
        isPointer: window.matchMedia('(pointer: fine)').matches,
      });
    };
    window.addEventListener('resize', handler, { passive: true });
    window.addEventListener('orientationchange', handler);
    return () => {
      window.removeEventListener('resize', handler);
      window.removeEventListener('orientationchange', handler);
    };
  }, []);

  return profile;
};
```

### 1.3 断点过渡策略

| 过渡场景 | 策略 | 实现方式 |
|---|---|---|
| PC → Pad横屏 | 侧边栏折叠，内容区自适应 | CSS Grid + `lg:grid-cols-[4rem_1fr]` |
| Pad横屏 → Pad竖屏 | 双栏变单栏，底部导航浮现 | `md:flex-col` + 导航栏定位切换 |
| Pad竖屏 → 手机横屏 | 保持单栏，元素紧凑排列 | Container Query + `min()` 函数 |
| 手机横屏 → 手机竖屏 | 双栏堆叠，Tab导航激活 | `max-sm:flex-col` + 条件渲染 |
| 任意方向切换 | 防抖重算，CSS过渡动画 | `transition-all duration-300 ease-in-out` |

## 2. 布局策略

### 2.0 全局布局骨架

```typescript
// src/components/layout/AppShell/AppShell.tsx
export const AppShell: React.FC = ({ children }) => {
  const { deviceType, orientation } = useDeviceProfile();

  return (
    <div className="h-screen w-screen flex overflow-hidden bg-nos-bg-primary"
         style={{ paddingTop: 'env(safe-area-inset-top)', paddingBottom: 'env(safe-area-inset-bottom)' }}>
      {/* PC端侧边栏 */}
      {deviceType === 'desktop' && <Sidebar variant="expanded" />}
      {/* Pad横屏折叠侧边栏 */}
      {deviceType === 'tablet' && orientation === 'landscape' && <Sidebar variant="collapsed" />}
      
      {/* 主内容区 */}
      <main className="flex-1 flex flex-col overflow-hidden">
        {/* Pad竖屏 + 手机顶部标题栏 */}
        {(deviceType === 'mobile' || (deviceType === 'tablet' && orientation === 'portrait')) && <MobileHeader />}
        
        {/* 页面内容 */}
        <div className="flex-1 overflow-y-auto overscroll-y-contain scroll-smooth">
          {children}
        </div>
        
        {/* 手机 + Pad竖屏底部导航 */}
        {(deviceType === 'mobile' || (deviceType === 'tablet' && orientation === 'portrait')) && <BottomNav />}
      </main>
    </div>
  );
};
```

```typescript
export const AdaptiveNav: React.FC = () => {
  const { deviceType, orientation } = useDeviceProfile();

  // PC端: 展开侧边栏
  if (deviceType === 'desktop') {
    return <SidebarExpanded width={240} collapsible />;
  }

  // Pad横屏: 折叠侧边栏
  if (deviceType === 'tablet' && orientation === 'landscape') {
    return <SidebarCollapsed width={64} />;
  }

  // Pad竖屏 + 手机: 底部Tab栏
  return <BottomTabBar items={navItems} />;
};
```

### 3.2 数据表格 → 卡片转换

```typescript
// src/components/data/AdaptiveTable/AdaptiveTable.tsx
export const AdaptiveTable = <T extends Record<string, any>>({
  columns,
  data,
  keyExtractor,
}: AdaptiveTableProps<T>) => {
  const { deviceType } = useDeviceProfile();
  const isMobile = deviceType === 'mobile';

  if (isMobile) {
    return (
      <div className="space-y-3">
        {data.map((item, index) => (
          <MobileDataCard
            key={keyExtractor(item, index)}
            item={item}
            columns={columns}
          />
        ))}
      </div>
    );
  }

  return (
    <div className="overflow-x-auto">
      <table className="w-full text-sm">
        <thead className="bg-nos-bg-secondary">
          <tr>
            {columns.map((col) => (
              <th key={col.key} className="px-4 py-3 text-left font-medium">
                {col.title}
              </th>
            ))}
          </tr>
        </thead>
        <tbody>
          {data.map((item, index) => (
            <tr key={keyExtractor(item, index)} className="border-b">
              {columns.map((col) => (
                <td key={col.key} className="px-4 py-3">
                  {col.render ? col.render(item[col.key], item) : item[col.key]}
                </td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
};

// 移动端数据卡片
const MobileDataCard = <T extends Record<string, any>>({
```

### 4.3 触摸手势定义

```typescript
// src/hooks/useTouchGestures.ts
export interface GestureConfig {
  onSwipeLeft?: () => void;
  onSwipeRight?: () => void;
  onSwipeUp?: () => void;
  onSwipeDown?: () => void;
  onPinchIn?: () => void;
  onPinchOut?: () => void;
  onLongPress?: () => void;
  onDoubleTap?: () => void;
  swipeThreshold?: number;   // 默认 50px
  longPressDelay?: number;   // 默认 500ms
}

export const useTouchGestures = (ref: RefObject<HTMLElement>, config: GestureConfig) => {
  const state = useRef({
    startX: 0, startY: 0, startTime: 0,
    lastTap: 0, isLongPress: false,
    initialDistance: 0,
  });

  const threshold = config.swipeThreshold ?? 50;
  const longPressDelay = config.longPressDelay ?? 500;

  useEffect(() => {
    const el = ref.current;
    if (!el) return;

    const onTouchStart = (e: TouchEvent) => {
      const touch = e.touches[0];
      state.current.startX = touch.clientX;
      state.current.startY = touch.clientY;
      state.current.startTime = Date.now();
      state.current.isLongPress = false;

      // 双指捏合检测
      if (e.touches.length === 2) {
        const dx = e.touches[0].clientX - e.touches[1].clientX;
        const dy = e.touches[0].clientY - e.touches[1].clientY;
        state.current.initialDistance = Math.sqrt(dx * dx + dy * dy);
      }

      // 长按检测
      if (config.onLongPress && e.touches.length === 1) {
        setTimeout(() => {
          state.current.isLongPress = true;
          config.onLongPress?.();
        }, longPressDelay);
      }
    };

    const onTouchEnd = (e: TouchEvent) => {
      const touch = e.changedTouches[0];
      const dx = touch.clientX - state.current.startX;
      const dy = touch.clientY - state.current.startY;
      const dt = Date.now() - state.current.startTime;

      if (state.current.isLongPress) return; // 长按已触发

      // 双击检测
      const now = Date.now();
      if (config.onDoubleTap && now - state.current.lastTap < 300) {
        config.onDoubleTap();
        state.current.lastTap = 0;
        return;
      }
      state.current.lastTap = now;

      // 滑动检测（时间短 + 位移大）
      if (dt < 300) {
        if (Math.abs(dx) > Math.abs(dy)) {
          if (dx > threshold) config.onSwipeRight?.();
          else if (dx < -threshold) config.onSwipeLeft?.();
        } else {
          if (dy > threshold) config.onSwipeDown?.();
          else if (dy < -threshold) config.onSwipeUp?.();
        }
      }
    };

    const onTouchMove = (e: TouchEvent) => {
      // 捏合检测
      if (e.touches.length === 2 && state.current.initialDistance > 0) {
        const dx = e.touches[0].clientX - e.touches[1].clientX;
        const dy = e.touches[0].clientY - e.touches[1].clientY;
        const distance = Math.sqrt(dx * dx + dy * dy);
        const scale = distance / state.current.initialDistance;

        if (scale > 1.2) config.onPinchOut?.();
        else if (scale < 0.8) config.onPinchIn?.();
      }
    };

    el.addEventListener('touchstart', onTouchStart, { passive: true });
    el.addEventListener('touchend', onTouchEnd);
    el.addEventListener('touchmove', onTouchMove, { passive: true });

    return () => {
      el.removeEventListener('touchstart', onTouchStart);
      el.removeEventListener('touchend', onTouchEnd);
      el.removeEventListener('touchmove', onTouchMove);
    };
  }, [ref, config, threshold, longPressDelay]);
};
```

### 4.5 拖拽操作跨设备实现

```typescript
// src/hooks/useAdaptiveDrag.ts
export const useAdaptiveDrag = <T extends HTMLElement>(
  ref: RefObject<T>,
  onDragEnd: (deltaX: number, deltaY: number) => void,
) => {
  const { isTouch } = useAdaptiveInteraction();
  const dragState = useRef({ isDragging: false, startX: 0, startY: 0 });

  useEffect(() => {
    const el = ref.current;
    if (!el) return;

    if (!isTouch) {
      // 鼠标拖拽
      const onMouseDown = (e: MouseEvent) => {
        dragState.current = { isDragging: true, startX: e.clientX, startY: e.clientY };
        el.style.cursor = 'grabbing';
        el.style.userSelect = 'none';
      };

      const onMouseMove = (e: MouseEvent) => {
        if (!dragState.current.isDragging) return;
        const dx = e.clientX - dragState.current.startX;
        const dy = e.clientY - dragState.current.startY;
        el.style.transform = `translate(${dx}px, ${dy}px)`;
      };

      const onMouseUp = (e: MouseEvent) => {
        if (!dragState.current.isDragging) return;
        const dx = e.clientX - dragState.current.startX;
        const dy = e.clientY - dragState.current.startY;
        dragState.current.isDragging = false;
        el.style.cursor = '';
        el.style.userSelect = '';
        el.style.transform = '';
        if (Math.abs(dx) > 5 || Math.abs(dy) > 5) {
          onDragEnd(dx, dy);
        }
      };

      el.addEventListener('mousedown', onMouseDown);
      window.addEventListener('mousemove', onMouseMove);
      window.addEventListener('mouseup', onMouseUp);

      return () => {
        el.removeEventListener('mousedown', onMouseDown);
        window.removeEventListener('mousemove', onMouseMove);
        window.removeEventListener('mouseup', onMouseUp);
      };
    } else {
      // 触摸拖拽
      const onTouchStart = (e: TouchEvent) => {
        if (e.touches.length !== 1) return;
        const touch = e.touches[0];
        dragState.current = {
          isDragging: true,
          startX: touch.clientX,
          startY: touch.clientY,
        };
      };

      const onTouchMove = (e: TouchEvent) => {
        if (!dragState.current.isDragging) return;
        e.preventDefault(); // 防止页面滚动
        const touch = e.touches[0];
        const dx = touch.clientX - dragState.current.startX;
        const dy = touch.clientY - dragState.current.startY;
        el.style.transform = `translate(${dx}px, ${dy}px)`;
      };

      const onTouchEnd = (e: TouchEvent) => {
        if (!dragState.current.isDragging) return;
        const touch = e.changedTouches[0];
        const dx = touch.clientX - dragState.current.startX;
        const dy = touch.clientY - dragState.current.startY;
        dragState.current.isDragging = false;
        el.style.transform = '';
        if (Math.abs(dx) > 10 || Math.abs(dy) > 10) {
          onDragEnd(dx, dy);
        }
      };

      el.addEventListener('touchstart', onTouchStart, { passive: true });
      el.addEventListener('touchmove', onTouchMove, { passive: false });
      el.addEventListener('touchend', onTouchEnd);

      return () => {
        el.removeEventListener('touchstart', onTouchStart);
        el.removeEventListener('touchmove', onTouchMove);
        el.removeEventListener('touchend', onTouchEnd);
      };
    }
  }, [ref, isTouch, onDragEnd]);
};
```

```typescript
  if ('fonts' in document) {
```

```typescript
export const optimizeTouchResponse = (): void => {
  // 1. 设置viewport阻止缩放延迟
  const viewport = document.querySelector('meta[name=viewport]');
  if (viewport) {
    viewport.setAttribute(
      'content',

    );

  // 2. 移除300ms点击延迟
  document.documentElement.style.touchAction = 'manipulation';

  // 3. 为所有滚动区域添加overscroll-behavior
  document.querySelectorAll('.scrollable').forEach((el) => {
```

```typescript
import { useVirtualizer } from '@tanstack/react-virtual';

interface VirtualListProps<T> {
  items: T[];
  itemHeight: number;
  renderItem: (item: T, index: number) => React.ReactNode;
  overscan?: number;
  containerHeight: number;
}

export const VirtualList = <T,>({
  items,
  itemHeight,
  renderItem,
  overscan = 5,
  containerHeight,
}: VirtualListProps<T>) => {
  const parentRef = useRef<HTMLDivElement>(null);

  const virtualizer = useVirtualizer({
    count: items.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => itemHeight,
    overscan,
  });

  return (
    <div
      ref={parentRef}
      className="overflow-auto"
      style={{ height: containerHeight, overscrollBehavior: 'contain' }}
    >
      <div
        style={{
          height: `${virtualizer.getTotalSize()}px`,
          width: '100%',
          position: 'relative',
        }}
      >
        {virtualizer.getVirtualItems().map((virtualItem) => (
          <div
            key={virtualItem.key}
            style={{
              position: 'absolute',
              top: 0,
              left: 0,
              width: '100%',
              height: `${itemHeight}px`,
              transform: `translateY(${virtualItem.start}px)`,
            }}
          >
            {renderItem(items[virtualItem.index], virtualItem.index)}
          </div>
        ))}
      </div>
    </div>
  );
};
```

### 6.4 WebSocket实时通信

```typescript
// src/api/websocket/WebSocketClient.ts
export class WebSocketClient {
  private ws: WebSocket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelay = 1000;
  private listeners: Map<string, Set<(data: any) => void>> = new Map();
  private pingInterval: NodeJS.Timer | null = null;

  constructor(private url: string) {}

  connect(): void {
    this.ws = new WebSocket(this.url);

    this.ws.onopen = () => {
      console.log('WebSocket connected');
      this.reconnectAttempts = 0;
      this.startPing();
    };

    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      this.handleMessage(message);
    };

    this.ws.onclose = () => {
      this.stopPing();
      this.attemptReconnect();
    };

    this.ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };
  }

  private handleMessage(message: WebSocketMessage): void {
    const { channel, data } = message;
    const callbacks = this.listeners.get(channel);
    callbacks?.forEach((cb) => cb(data));
  }

  subscribe(channel: string, callback: (data: any) => void): () => void {
    if (!this.listeners.has(channel)) {
      this.listeners.set(channel, new Set());
      this.send({ type: 'subscribe', channel });
    }
    this.listeners.get(channel)!.add(callback);

    return () => {
      this.listeners.get(channel)?.delete(callback);
      if (this.listeners.get(channel)?.size === 0) {
        this.send({ type: 'unsubscribe', channel });
        this.listeners.delete(channel);
      }
    };
  }

  send(message: object): void {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    }
  }

  private startPing(): void {
    this.pingInterval = setInterval(() => {
      this.send({ type: 'ping' });
    }, 30000);
  }

  private stopPing(): void {
    if (this.pingInterval) {
      clearInterval(this.pingInterval);
      this.pingInterval = null;
    }
  }

  private attemptReconnect(): void {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) return;
    this.reconnectAttempts++;
    const delay = this.reconnectDelay * Math.pow(2, this.reconnectAttempts - 1);
    setTimeout(() => this.connect(), Math.min(delay, 30000));
  }

  disconnect(): void {
    this.stopPing();
    this.ws?.close();
  }
}

// React Hook封装
export const useWebSocket = (channel: string) => {
  const [data, setData] = useState<any>(null);
  const [connected, setConnected] = useState(false);
  const wsClient = useRef<WebSocketClient>();

  useEffect(() => {
    const client = new WebSocketClient(import.meta.env.VITE_WS_URL);
    wsClient.current = client;

    client.connect();

    const unsubscribe = client.subscribe(channel, (newData) => {
      setData(newData);
      setConnected(true);
    });

    return () => {
      unsubscribe();
      client.disconnect();
    };
  }, [channel]);

  const send = useCallback((message: object) => {
    wsClient.current?.send(message);
  }, []);

  return { data, connected, send };
};
```

```tsx
// src/components/Image/ResponsiveImage.tsx
export const ResponsiveImage: React.FC<ResponsiveImageProps> = ({
  src,
  alt,
  aspectRatio,
  className,
  objectFit = 'cover',
  sizes,
}) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const { width } = useResizeObserver(containerRef);
  const [imgLoaded, setImgLoaded] = useState(false);

  // 根据容器宽度生成最佳图片URL
  const optimizedSrc = useMemo(() => {
    if (width === 0) return `${src}?w=100&q=80&blur=20`; // 初始模糊占位
    const targetWidth = Math.min(width * (window.devicePixelRatio || 1), 1920);
    return `${src}?w=${Math.round(targetWidth)}&q=85&format=webp`;
  }, [src, width]);

  // srcset 用于不同DPI屏幕
  const srcSet = useMemo(() => {
    if (!sizes) return undefined;
    return sizes.map(({ width: w, descriptor }) =>
      `${src}?w=${w}&q=85&format=webp ${descriptor}`
    ).join(', ');
  }, [src, sizes]);

  return (
    <div
      ref={containerRef}
      className={`relative overflow-hidden bg-gray-100 ${className}`}
      style={{ aspectRatio }}
    >
      {/* 骨架屏占位 */}
      {!imgLoaded && (
        <div className="absolute inset-0 animate-shimmer"
             style={{
               background: 'linear-gradient(90deg, #f0f0f0 25%, #e0e0e0 50%, #f0f0f0 75%)',
               backgroundSize: '200% 100%',
             }}
        />
      )}
      <img
        src={optimizedSrc}
        srcSet={srcSet}
        alt={alt}
        loading="lazy"
        decoding="async"
        onLoad={() => setImgLoaded(true)}
        className={`w-full h-full transition-opacity duration-500
                    ${objectFit === 'cover' ? 'object-cover' : 'object-contain'}
                    ${imgLoaded ? 'opacity-100' : 'opacity-0'}`}
        style={{ willChange: 'opacity' }}
      />
    </div>
  );
};
```

```tsx
// src/components/layout/SafeAreaProvider/SafeAreaProvider.tsx
export const SafeAreaProvider: React.FC<{ children: React.ReactNode }> = ({
  children,
}) => {
  // 设置CSS变量存储安全区信息
  useEffect(() => {
    // iOS 15+ 支持 env() CSS函数，但需要viewport-fit=cover
    const viewportMeta = document.querySelector('meta[name="viewport"]');
    if (viewportMeta) {
      const content = viewportMeta.getAttribute('content') || '';
      if (!content.includes('viewport-fit=cover')) {
        viewportMeta.setAttribute(
          'content',
          `${content}, viewport-fit=cover`
        );
      }
    }

    // 检测刘海屏/灵动岛并设置CSS变量
    const setSafeAreaVars = () => {
      const safeTop = getComputedStyle(document.documentElement)
        .getPropertyValue('env(safe-area-inset-top)');
      const safeBottom = getComputedStyle(document.documentElement)
        .getPropertyValue('env(safe-area-inset-bottom)');

      document.documentElement.style.setProperty('--sat', safeTop || '0px');
      document.documentElement.style.setProperty('--sab', safeBottom || '0px');
    };

    setSafeAreaVars();
    window.addEventListener('orientationchange', setSafeAreaVars);
    return () => window.removeEventListener('orientationchange', setSafeAreaVars);
  }, []);

  return <>{children}</>;
};

// 安全区底部填充组件（用于底部导航/操作栏上方）
export const SafeAreaBottom: React.FC<{ className?: string }> = ({
  className,
}) => {
  return (
    <div
      className={className}
      style={{ height: 'env(safe-area-inset-bottom, 0px)' }}
    />
  );
};

// 安全区域Hook
export const useSafeArea = () => {
  const [safeArea, setSafeArea] = useState({
    top: 0,
    bottom: 0,
    left: 0,
    right: 0,
  });

  useEffect(() => {
    const updateSafeArea = () => {
      // 通过创建临时元素获取计算后的安全区值
      const div = document.createElement('div');
      div.style.cssText = `
        position: fixed;
        padding-top: env(safe-area-inset-top);
        padding-bottom: env(safe-area-inset-bottom);
        padding-left: env(safe-area-inset-left);
        padding-right: env(safe-area-inset-right);
      `;
      document.body.appendChild(div);
      const computed = getComputedStyle(div);
      setSafeArea({
        top: parseInt(computed.paddingTop) || 0,
        bottom: parseInt(computed.paddingBottom) || 0,
        left: parseInt(computed.paddingLeft) || 0,
        right: parseInt(computed.paddingRight) || 0,
      });
      document.body.removeChild(div);
    };

    updateSafeArea();
    window.addEventListener('orientationchange', updateSafeArea);
    return () => window.removeEventListener('orientationchange', updateSafeArea);
  }, []);

  return safeArea;
```

```typescript
export const isSafari = (): boolean =>
  /^((?!chrome|android).)*safari/i.test(navigator.userAgent);

export const isPWA = (): boolean =>
  window.matchMedia('(display-mode: standalone)').matches ||
  (window.navigator as any).standalone === true;

export const isIPad = (): boolean =>
  /iPad/.test(navigator.userAgent) ||
  (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);

export const getIPadModel = (): string => {
  if (!isIPad()) return 'not-ipad';
  const width = window.screen.width;
  const height = window.screen.height;
  const ratio = window.devicePixelRatio;
  if (width === 1024 && height === 1366) return 'iPad Pro 12.9"';
  if (width === 834 && height === 1194) return 'iPad Pro 11"';
  if (width === 820 && height === 1180) return 'iPad Air 10.9"';
  if (width === 810 && height === 1080) return 'iPad 10.2"';
  if (width === 744 && height === 1133) return 'iPad Mini 8.3"';
  return `iPad (${width}x${height} @${ratio}x)`;
};
```
